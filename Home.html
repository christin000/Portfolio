// Toy runtime: async task scheduler
struct Task { id: u64, work: Box<dyn FnOnce() + Send> }
struct Scheduler { q: VecDeque<Task> }

impl Scheduler {
    fn spawn(&mut self, t: Task) { self.q.push_back(t) }
    fn run(&mut self) {
        while let Some(t) = self.q.pop_front() {
            (t.work)(); // cooperative
        }
    }
}

fn main() {
    let mut s = Scheduler { q: VecDeque::new() };
    for id in 0..3 { s.spawn(Task { id, work: Box::new(|| println!("task {}", id)) }) }
    s.run();
}
